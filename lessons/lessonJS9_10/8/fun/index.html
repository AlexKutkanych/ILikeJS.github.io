Перенести действие в аргумент?!

  function myForEach(array, action ) {
    for (var i = 0; i < array.length; i++) {
      action (array[i]);
    }
  }
  myForEach([3,9,27], (a) => console.log (a = a + 1));
  //4; 10; 28
